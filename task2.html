Task 2

Write a simple program to combine a first name and last name inside a function. Then update the function to accept a first and last name as arguments.

Please enter your First Name
Please enter your Surname
function setName(firstName, lastName) {
  document.getElementById('fullName').innerHTML = `${firstName} ${lastName}`;
}

function buildFullName() {
  var firstName = document.getElementById('firstName').value;
  var lastName = document.getElementById('surname').value;
  // Just return if either are blank. There will be a validation message
  if (firstName == "" || lastName == "") {
      return;
  }

  setName(firstName, lastName);
}
